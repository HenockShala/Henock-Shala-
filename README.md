  function getCookies() {
    let cookies = document.cookie.split(';');
    let cookieList = {};
    cookies.forEach(cookie => {
        let [name, value] = cookie.split('=');
        name = name.trim();
        if (name) {
            cookieList[name] = value ? decodeURIComponent(value.trim()) : "";
        }
    });
    return cookieList;
}
